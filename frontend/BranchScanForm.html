API Reference — POST /api/v1/branch/scan

// This page calls the backend using FormData (NOT JSON)
// because we are sending a file along with the fields

const formData = new FormData()
formData.append('branch_code',             'MUM-001')
formData.append('branch_name',             'Andheri West Branch')
formData.append('staff_id',                'RM-001')
formData.append('staff_name',              'Sahil Gupta')
formData.append('kyc_physically_seen',     true)
formData.append('customer_consent_signed', false)
formData.append('file',                    fileInput.files[0])

// DO NOT set Content-Type header manually
// Browser sets it automatically with the correct multipart boundary

fetch('/api/v1/branch/scan', {
  method: 'POST',
  body: formData
})
.then(res => res.json())
.then(data => {
  console.log(data.application_id)   // e.g. "a1b2c3d4-..."
  console.log(data.status)           // "pending"
  console.log(data.message)          // "Scan received. OCR in progress."
})
